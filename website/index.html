use riglr_agents::agents::tool_calling::ToolCallingAgentBuilder;
use riglr_agents::toolset::Toolset;
use riglr_agents::*;
use riglr_core::provider::ApplicationContext;
use riglr_core::signer::{SignerContext, UnifiedSigner};
use riglr_solana_tools::signer::LocalSolanaSigner;
use serde_json::json;
use solana_sdk::signature::{Keypair, Signer};
use std::sync::Arc;

// 1. Load environment and setup context
let app_context = ApplicationContext::from_env();
app_context.set_extension(Arc::new(
    solana_client::rpc_client::RpcClient::new(devnet_rpc)
));

// 2. Discover all available tools with one line
let toolset = Toolset::new().with_solana_tools();

// 3. Build the entire agent with one clean, fluent call
let openai_client = rig::providers::openai::Client::new(&openai_api_key);
let live_agent = ToolCallingAgentBuilder::new(toolset, app_context)
    .build(openai_client.agent("gpt-4o"))
    .await?;

// 4. Setup signer and execute operations
let solana_signer = LocalSolanaSigner::new(signer_keypair, devnet_rpc);
let unified_signer: Arc = Arc::new(solana_signer);

SignerContext::with_signer(unified_signer, async {
    let registry = LocalAgentRegistry::new();
    registry.register_agent(live_agent).await.unwrap();
    let dispatcher = AgentDispatcher::new(Arc::new(registry));
    
    // READ: Query on-chain data  
    let read_task = Task::new(
        TaskType::Custom("tool_calling".to_string()),
        json!({ "prompt": "What is the SOL balance of alice.sol?" })
    );
    let result = dispatcher.dispatch_task(read_task).await?;
    
    // WRITE: Execute transactions
    let write_task = Task::new(
        TaskType::Custom("tool_calling".to_string()),
        json!({ "prompt": "Send 0.001 SOL to bob.sol" })
    );
    let tx_result = dispatcher.dispatch_task(write_task).await?;
}).await?;

How riglr Takes rig to Production

While rig provides the powerful "brain" for LLM-to-tool orchestration, riglr adds the complete "body" and "nervous system"
required for agents to operate securely, reliably, and at scale in the complex blockchain environment.

1

SignerContext: Transactional Security

Solve the hardest problem in blockchain AI: secure key management

The Problem: How can an LLM-driven agent sign transactions without exposing private keys to the reasoning loop? How do you ensure multi-tenant safety?

The riglr Solution: Our SignerContext uses task-local storage to scope signers to specific async tasks. Keys are never passed as parameters, they're available only within secure with_signer blocks, and automatically cleaned up on exit. Each concurrent request gets complete isolation.

2

ApplicationContext: Clean Dependency Injection

Decouple tools from concrete implementations

The Problem: How can tools access shared resources (RPC clients, API keys, databases) without hardcoding or creating circular dependencies?

The riglr Solution: The ApplicationContext holds shared resources wrapped in Arc. Tools retrieve dependencies as extensions, enforcing a clean dependency graph. This enables easy testing with mock clients and flexible configuration for any number of chains or services.

3

#[tool] Macro: Superior Developer Experience

Eliminate boilerplate while maintaining type safety

The Problem: Manually implementing rig's Tool trait requires verbose boilerplate for serialization, schema generation, and error handling.

The riglr Solution: Our #[tool] macro automates everything: generates serde-compatible args structs, derives JSON schemas, integrates doc comments as descriptions, and handles error conversion. Write async functions, get production-ready tools.

4

Two-Level Error Handling

Intelligent error classification for blockchain operations

The Problem: Blockchain operations fail for many reasons - some temporary (network timeouts), others permanent (invalid address). Simple errors can't distinguish.

The riglr Solution: High-level ToolError classifies failures as Retriable, Permanent, RateLimited, or InvalidInput. The ToolWorker implements automatic retry with exponential backoff. Chain-specific errors provide detailed context when needed.

Multi-Agent Coordination (riglr-agents)

Create "swarms" of specialized agents that collaborate on complex tasks:

  • Agent Specialization: ResearchAgent, RiskAgent, ExecutionAgent
  • Registry System: Local or Redis-based discovery
  • Smart Dispatcher: Capability, LeastLoaded, RoundRobin
  • Inter-Agent Communication: Built-in messaging
// Complex workflow with specialized agents
let research_task = Task::new(
    TaskType::Research, 
    json!({ "query": "Analyze SOL price trends" })
);
let research = dispatcher.dispatch_task(research_task).await?;

let risk_task = Task::new(
    TaskType::RiskAnalysis, 
    research.data()
);
let risk = dispatcher.dispatch_task(risk_task).await?;

if risk.is_approved() {
    let execution_task = Task::new(
        TaskType::Trading,
        json!({ "action": "buy", "amount": 100 })
    );
    dispatcher.dispatch_task(execution_task).await?;
}

Real-Time Event Streaming (riglr-streams)

Make agents proactive rather than reactive:

  • Multi-Source: Solana Geyser, EVM WebSockets, Binance
  • Resilient: Auto-reconnect, circuit breakers, backoff
  • Composable: map, filter, throttle, debounce, batch
  • Event-Triggered: Auto-execute on conditions
// Complex pipeline with fluent API
let solana_stream = SolanaGeyserStream::new(...);
let processed = solana_stream
    .filter(|e| matches!(e.kind(), EventKind::Swap))
    .map(extract_swap_metrics)
    .throttle(Duration::from_secs(10))
    .batch(50, Duration::from_secs(60));

// React to real-time events
processed.subscribe(|batch| async {
    if batch.volume > threshold {
        agent.execute_strategy(batch).await?;
    }
}).await;

Enterprise Architecture Patterns

riglr establishes robust, chain-agnostic patterns that form the foundation for production blockchain applications.

üîê Thread-Local Signer Isolation

The SignerContext pattern ensures complete isolation between concurrent users. Each async task gets its own signer scope, preventing cross-contamination in multi-tenant environments. Keys are never exposed to the LLM reasoning loop.

üéØ Dependency Injection Container

The ApplicationContext decouples tools from concrete implementations. RPC clients, API keys, and databases are injected as extensions, enabling clean testing with mocks and supporting multiple chains without code changes.

üîÑ Behavioral Error Classification

Two-level error handling separates high-level behavior (Retriable, Permanent, RateLimited) from low-level details (BlockhashExpired, InsufficientFunds). The ToolWorker uses this classification for automatic retry with exponential backoff.

‚öôÔ∏è Unified Configuration System

Hierarchical, environment-driven config with fail-fast validation. The RPC_URL_{CHAIN_ID} convention allows adding new EVM chains via environment variables without any code changes.

Simplified Multi-Tenant Security

The Challenge

Blockchain applications need to manage multiple user wallets safely while preventing cross-user access. This typically requires complex key management and careful isolation.

How riglr Helps

riglr provides a SignerContext that uses thread-local storage to isolate user operations. Tools can access the appropriate signer without explicit key passing, reducing boilerplate and potential errors. 

// Tools are completely stateless
#[tool]
async fn transfer_sol(
    to: String,
    amount: f64,
) -> Result {
    // SignerContext automatically provides
    // the correct user's signer
    let signer = get_signer_from_context()?;
    let tx = create_transfer(&signer, &to, amount).await?;
    Ok(tx.signature.to_string())
}

// No manual key management needed!
// Each user request runs in isolated context

‚úÖ After riglr
/// Get the SOL balance for a given wallet address.
#[tool]
async fn get_sol_balance(
    /// The Solana wallet address (base58 encoded).
    address: String,
) -> Result<f64, ToolError> {
    // Your logic here. That's it.
    // The macro handles the rest.
    let client = get_client_from_context()?;
    let lamports = client.get_balance(&address).await?;
    Ok(lamports as f64 / 1_000_000_000.0)
}
