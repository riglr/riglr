use riglr_agents::agents::tool_calling::ToolCallingAgentBuilder;
use riglr_agents::toolset::Toolset;
use riglr_agents::*;
use riglr_core::provider::ApplicationContext;
use riglr_core::signer::{SignerContext, UnifiedSigner};
use riglr_solana_tools::signer::LocalSolanaSigner;
use serde_json::json;
use solana_sdk::signature::{Keypair, Signer};
use std::sync::Arc;

// 1. Load environment and setup context
let app_context = ApplicationContext::from_env();
app_context.set_extension(Arc::new(
    solana_client::rpc_client::RpcClient::new(devnet_rpc)
));

// 2. Discover all available tools with one line
let toolset = Toolset::new().with_solana_tools();

// 3. Build the entire agent with one clean, fluent call
let openai_client = rig::providers::openai::Client::new(&openai_api_key);
let live_agent = ToolCallingAgentBuilder::new(toolset, app_context)
    .build(openai_client.agent("gpt-4o"))
    .await?;

// 4. Setup signer and execute operations
let solana_signer = LocalSolanaSigner::new(signer_keypair, devnet_rpc);
let unified_signer: Arc = Arc::new(solana_signer);

SignerContext::with_signer(unified_signer, async {
    let registry = LocalAgentRegistry::new();
    registry.register_agent(live_agent).await.unwrap();
    let dispatcher = AgentDispatcher::new(Arc::new(registry));
    
    // READ: Query on-chain data  
    let read_task = Task::new(
        TaskType::Custom("tool_calling".to_string()),
        json!({ "prompt": "What is the SOL balance of alice.sol?" })
    );
    let result = dispatcher.dispatch_task(read_task).await?;
    
    // WRITE: Execute transactions
    let write_task = Task::new(
        TaskType::Custom("tool_calling".to_string()),
        json!({ "prompt": "Send 0.001 SOL to bob.sol" })
    );
    let tx_result = dispatcher.dispatch_task(write_task).await?;
}).await?;

Battle-tested architectural patterns from real deployments that solve the hardest problems

üîê SignerContext: Thread-Local Transaction Signing

Keys NEVER exposed to LLM ‚Ä¢ Complete multi-tenant isolation

// Each user request gets completely isolated signer context
SignerContext::with_signer(user_signer, async {
    // Keys NEVER exposed to LLM reasoning loop
    // Automatic cleanup when scope exits
    // Zero cross-tenant contamination risk
    agent.execute("Send 0.1 SOL to alice.sol").await
}).await

Why It Matters: In production, multiple users make concurrent requests. Each gets their own isolated signer context with automatic cleanup. Keys are never passed as parameters, preventing exposure to the LLM.

2

üéØ ApplicationContext: Clean Dependency Injection

Modular architecture ‚Ä¢ Mock-friendly testing ‚Ä¢ Zero circular dependencies

// Application injects dependencies once
context.set_extension(Arc::new(solana_client));
context.set_extension(Arc::new(evm_provider));

// Tools retrieve what they need - no hardcoding
#[tool]
async fn get_balance(address: String) -> Result {
    let client = context.get_extension::()?;
    // Use client...
}

Why It Matters: Decouples tools from implementations, enables mock testing without code changes, supports multiple chains without circular dependencies.

3

üîÑ Two-Level Error Handling with Auto-Retry

Behavioral classification ‚Ä¢ Exponential backoff ‚Ä¢ Chain-specific details

// High-level behavioral classification
match error {
    ToolError::Retriable(_) => // Automatic exponential backoff
    ToolError::RateLimited(_) => // Respect rate limits
    ToolError::Permanent(_) => // Don't retry
}

// Low-level chain-specific details preserved
if let Some(solana_error) = error.downcast::() {
    match solana_error {
        SolanaToolError::BlockhashExpired => // Handle specifically
    }
}

Why It Matters: Automatic retry logic for transient failures, intelligent rate limit handling, detailed error context for debugging.

Multi-Agent Coordination (riglr-agents)

Create "swarms" of specialized agents that collaborate on complex tasks:

  • Agent Specialization: ResearchAgent, RiskAgent, ExecutionAgent
  • Registry System: Local or Redis-based discovery
  • Smart Dispatcher: Capability, LeastLoaded, RoundRobin

// Complex workflow with specialized agents
let research_task = Task::new(
    TaskType::Research, 
    json!({ "query": "Analyze SOL price trends" })
);
let research = dispatcher.dispatch_task(research_task).await?;

let risk_task = Task::new(
    TaskType::RiskAnalysis, 
    research.data()
);
let risk = dispatcher.dispatch_task(risk_task).await?;

if risk.is_approved() {
    let execution_task = Task::new(
        TaskType::Trading,
        json!({ "action": "buy", "amount": 100 })
    );
    dispatcher.dispatch_task(execution_task).await?;
}

Real-Time Event Streaming (riglr-streams)

Make agents proactive rather than reactive:

  • Multi-Source: Solana Geyser, EVM WebSockets, Binance
  • Resilient: Auto-reconnect, circuit breakers, backoff
  • Composable: map, filter, throttle, debounce, batch
  • Event-Triggered: Auto-execute on conditions
// Complex pipeline with fluent API
let solana_stream = SolanaGeyserStream::new(...);
let processed = solana_stream
    .filter(|e| matches!(e.kind(), EventKind::Swap))
    .map(extract_swap_metrics)
    .throttle(Duration::from_secs(10))
    .batch(50, Duration::from_secs(60));

// React to real-time events
processed.subscribe(|batch| async {
    if batch.volume > threshold {
        agent.execute_strategy(batch).await?;
    }
}).await;

/// Get the SOL balance for a given wallet address.
#[tool]
async fn get_sol_balance(
    /// The Solana wallet address (base58 encoded).
    address: String,
) -> Result<f64, ToolError> {
    let client = get_client_from_context()?;
    let lamports = client.get_balance(&address).await?;
    Ok(lamports as f64 / 1_000_000_000.0)
}
